import { Injectable } from '@angular/core';

import { FoodItem } from '../../../../globalModels/foodItem'
import { Order } from '../../../../globalModels/order'
import { Pantry } from '../../../../globalModels/pantry'
import { FirebaseService } from './firebase.service';
import { OrderStatus } from '../../../../globalModels/orderStatus';


interface TimestampedOrders {
  orders: number
  month: string
}
interface TimestampedItems {
  items: number
  month: string
}
@Injectable({
  providedIn: 'root'
})

export class ForecastService {

  pantry: Pantry
  inventory: FoodItem[]
  orders: Order[]

  constructor() {

  }

  setData(dataPantry: Pantry, dataInventory: FoodItem[], dataOrders: Order[]) {
    this.pantry = dataPantry
    this.inventory = dataInventory
    this.orders = dataOrders
  }
  /**
   * @brief Returns the total numbers of orders for a given pantry
   * @param type optional OrderStatus stating the desired OrderStatus
   * @returns number of orders, optionally of a type
   */
  getNumOrder(type?: OrderStatus): number {
    let retval = 0
    if (type) {
      this.orders.forEach(order => {
        if (order.status === type) {
          retval += 1
        }
      });
    }
    else if (this.orders) {
      retval = this.orders?.length
    }
    return retval
  }

  /**
   * @brief Gets a list of all food items of order types not cancelled or picked up
   * @returns FoodItem list of all active items
   */
  getOutstandingItems(): FoodItem[] {
    let retval: FoodItem[] = []
    this.orders.forEach(order => {
      if (order.status === OrderStatus.Processing) {
        retval = retval.concat(order.items)
      }
    });
    return retval
  }

  /**
   * @brief Sorts a list of orders by datetime
   * @param orders The orderlist to be sorted
   */
  sortOrdersByDate(orders: Order[]) {
    orders.sort((a: Order, b: Order) => {
      return (- a.timestamp.getTime() + b.timestamp.getTime())
    })
  }

  /**
   * @brief Returns a list of orders that have been filled 
   * @returns Order[] of filled orders
   */
  getOrderHistory(): Order[] {
    let retval: Order[] = []
    this.orders.forEach(order => {
      if (order.status === OrderStatus['Order Picked Up'])
        retval.push(order)
    });
    this.sortOrdersByDate(retval)
    return retval
  }

  /**
   * @brief For a list of food items, returns the number of occurences for that item
   * @param list The list of food items
   * @param itemName The item name
   * @returns number of occurences in the foodlist
   */
  getNumItemOccurrences(list: FoodItem[], itemName: string): number {
    let counter: number = 0
    list.forEach(item => {
      if (item.name === itemName) {
        counter += item.quantity
      }
    });
    return counter
  }

  /**
   * @brief Creates a list of all food items in a given order list
   * @param orders The orders to parse
   * @returns The full list of items from an order list
   */
  getItemsFromOrders(orders: Order[]) {
    let retval: FoodItem[] = []
    orders.forEach(order => {
      retval = retval.concat(order.items)
    });
    return retval
  }

  /**
   * @brief Returns a list of lists where sublist are the orders for a given month
   * @returns History list of lists 
   */
  orderHistoryByMonth() {
    let retval: Order[][] = []
    let history = this.getOrderHistory()
    let i: number = 0
    this.sortOrdersByDate(history)

    history.forEach(order => {
      if (retval.length === 0) {
        retval.push([order])
      }
      else if (order.timestamp.getMonth() === retval[i][0].timestamp.getMonth()
        && order.timestamp.getFullYear() === retval[i][0].timestamp.getFullYear()) {
        retval[i].push(order)
      }
      else {
        retval.push([order])
        i += 1
      }
    });
    return retval.reverse()
  }

  /**
   * @brief Reshapes an order list to a list of order lists by month
   * @param orderList The list of orders
   * @returns List of order lists by month
   */
  sortOrderListByMonth(orderList: Order[]) {
    let retval: Order[][] = []
    let i: number = 0
    this.sortOrdersByDate(orderList)

    orderList.forEach(order => {
      if (retval.length === 0) {
        retval.push([order])
      }
      else if (order.timestamp.getMonth() === retval[i][0].timestamp.getMonth()
        && order.timestamp.getFullYear() === retval[i][0].timestamp.getFullYear()) {
        retval[i].push(order)
      }
      else {
        retval.push([order])
        i += 1
      }
    });
    return retval
  }

  /**
   * @brief Returns a list containing the number of orders by month
   * @returns List of numbers of orders by month
   */
  getNumOrderByMonth() {
    let retval: TimestampedOrders[] = []
    let history = this.orderHistoryByMonth()
    history.forEach(orderList => {
      retval.push({
        "orders": orderList.length,
        "month": orderList[0].timestamp.toLocaleString('default', {month: 'long', year: 'numeric'})
      })
    });
    return retval
  }

  /**
   * @brief Returns a list of the number of item occurences by month
   * @param itemName The name of the item to query
   */
  getItemDemandByMonth(itemName: string) {
    let retval: TimestampedItems[] = []
    let history = this.orderHistoryByMonth()
    history.forEach(orderList => {
      let sum: number = 0
      orderList.forEach(order => {
        sum += this.getNumItemOccurrences(order.items, itemName)
      });
      retval.push({
        "month": orderList[0].timestamp.toLocaleString('default', {month: 'long', year: 'numeric'}),
        "items": sum
      })
    })
    return retval
  }

  /**
   * @brief Returns a list of timestamped items within a date range
   * @param itemName The name of the item to search for
   * @param start The start date of the range
   * @param end The end date of the range
   * @returns List of timestamped items
   */
  getItemDemandInRange(itemName:string, start:Date, end:Date) {
    let retval: TimestampedItems[] = []
    let history = this.getOrderHistory()

    let cut = this.getOrdersInDateRange(history, start, end)
    let cutReshaped = this.sortOrderListByMonth(cut)

    cutReshaped.forEach(orderList => {
      let sum: number = 0
      orderList.forEach(order => {
        sum += this.getNumItemOccurrences(order.items, itemName)
      });
      retval.push({
        "month": orderList[0].timestamp.toLocaleString('default', {month: 'long', year: 'numeric'}),
        "items": sum
      })
    })
    return retval
  }

  /**
   * @brief Returns the list of orders within a given range
   * @param orderList The list of orders to parse
   * @param start The start date of the range
   * @param end The end date of the range
   * @returns The list of orders falling within the range
   */
  getOrdersInDateRange(orderList: Order[], start: Date, end: Date) {
    let retval: Order[] = []
    orderList.forEach(order => {
      if (order.timestamp.getTime() >= start.getTime() && order.timestamp.getTime() <= end.getTime()) {
        retval.push(order)
      }
    });
    return retval
  }

  /**
   * @brief Calculates the date object n months before the given date
   * @param now The starting date
   * @param n The number of months to regress
   * @returns The date n months from the given date
   */
  calcPrevDate(now: Date, n: number): Date {
    let then = new Date(now.toDateString())
    then.setMonth(now.getMonth() - n)
    return then
  }

  /**
   * @brief Returns the previous years order count for the same month
   * @returns The previous years monthly demand
   */
  getPreviousYearDemand(): Number {
    let history = this.getOrderHistory()
    let now = new Date()
    let start = this.calcPrevDate(now, 13)
    let end = this.calcPrevDate(now, 12)
    let cut = this.getOrdersInDateRange(history, start, end)
    return cut.length
  }

  /**
   * @brief Returns the previous years item demand within the same month
   * @param itemName The name of the item
   * @returns The total number of item occurences for the previous year during the same month
   */
  getItemPreviousYearDemand(itemName: string): Number {
    let history = this.getOrderHistory()
    let now = new Date()
    let start = this.calcPrevDate(now, 13)
    let end = this.calcPrevDate(now, 12)
    let cut = this.getOrdersInDateRange(history, start, end)
    let items = this.getItemsFromOrders(cut)
    return this.getNumItemOccurrences(items, itemName)
  }

  /**
   * Applies linear regression to find the slope of a list of numbers
   * @param list a list of numbers
   * @returns The slope of the linear regression
   */
  calcLinearRegression(list: number[]) {
    let n = list.length

    if (n == 0) {
      return 0
    }
    else {

      let a: number, b: number, c: number, d: number, i: number
      a = b = c = d = i = 0
      list.forEach(element => {
        i++
        a += i * element
        b += i
        c += element
        d += i ** 2
      });

      return (n * (a) - b * c) / (n * d - (b ** 2))
    }
  }

  //FORECASTING AND PREDICTION FUNCTIONS

  /**
   * @brief Computes the current month order demand percent difference to previous year order demand
   * @returns The percent difference in current year order demand to previous year order demand
   */
  prevYearDiff() {
    let history = this.getOrderHistory()
    let t2 = new Date()
    let t1 = this.calcPrevDate(t2, 1)
    let t3 = this.calcPrevDate(t2, 13)
    let t4 = this.calcPrevDate(t2, 12)
    let period1 = this.getOrdersInDateRange(history, t1, t2)
    let period2 = this.getOrdersInDateRange(history, t3, t4)

    if (period2.length == 0) {
      return 0
    }

    return (period1.length - period2.length) / period2.length
  }

  /**
   * @brief Computes the current month item demand percent difference compared to previous year
   * @param itemName The name of the item to parse
   * @returns The percent difference of current to previous year item demand
   */
  prevYearItemDiff(itemName: string) {
    let history = this.getOrderHistory()
    let t2 = new Date()
    let t1 = this.calcPrevDate(t2, 1)
    let t3 = this.calcPrevDate(t2, 13)
    let t4 = this.calcPrevDate(t2, 12)
    let period1 = this.getOrdersInDateRange(history, t1, t2)
    let period2 = this.getOrdersInDateRange(history, t3, t4)

    let currentDemand = this.getNumItemOccurrences(this.getItemsFromOrders(period1), itemName)
    let previousDemand = this.getNumItemOccurrences(this.getItemsFromOrders(period2), itemName)

    if (previousDemand == 0) {
      return 0
    }

    return (currentDemand - previousDemand) / previousDemand
  }

  /**
   * @brief Returns The linear regression of total order frequency trend on a monthly basis for n months
   * @param n The number of months n to compute the linear regression on
   * @returns A trend, positive or negative, corresponding to the desired period
   */
  getXMonthTrend(n: number): number {
    let history = this.getOrderHistory()
    let now = new Date()
    let then = this.calcPrevDate(now, n)
    let cut = this.getOrdersInDateRange(history, then, now)
    let i: number = 0

    let buckets: number[] = []
    let curMonth: number = 0

    cut.forEach(order => {
      if (buckets.length == 0) {
        buckets = [1]
        curMonth = order.timestamp.getMonth()
      }
      else if (order.timestamp.getMonth() == curMonth) {
        buckets[i] += 1
      }
      else {
        buckets.push(1)
        curMonth = order.timestamp.getMonth()
        i++
      }
    });

    // Negation a result of most recent months occurring at front of array
    return -this.calcLinearRegression(buckets)
  }

  /**
   * @brief Returns The linear regression of item frequency trend on a monthly basis for n months
   * @param n The number of months n to compute the linear regression on
   * @param itemName The name of the item to parse for
   * @returns A trend, positive or negative, corresponding to the desired period
   */
  getXMonthItemTrend(n: number, itemName: string): number {
    let history = this.getOrderHistory()
    let now = new Date()
    let then = this.calcPrevDate(now, n)
    let cut = this.getOrdersInDateRange(history, then, now)
    let i: number = 0

    let buckets: number[] = []
    let curMonth: number = 0

    cut.forEach(order => {
      if (buckets.length == 0) {
        buckets = [this.getNumItemOccurrences(order.items, itemName)]
        curMonth = order.timestamp.getMonth()
      }
      else if (order.timestamp.getMonth() == curMonth) {
        buckets[i] += this.getNumItemOccurrences(order.items, itemName)
      }
      else {
        buckets.push(this.getNumItemOccurrences(order.items, itemName))
        curMonth = order.timestamp.getMonth()
        i++
      }
    });
    // Negation a result of most recent months occurring at front of array
    return -this.calcLinearRegression(buckets)
  }

  /**
   * @brief Computes the 1 Dimensional Convolution of an array with a kernel
   * @note Uses average padding
   * @param array The array to be convolved
   * @param kernel The kernel to convolve with
   * @param norm The normalization factor of the convolution
   * @returns The result of the convolution, an array of numbers
   */
  convolve1D(array: number[], kernel: number[], norm: number): number[] {
    let copy = []
    let retval = []
    let avg = 0

    array.forEach(num => {
      copy.push(num)
      avg += num
    });

    avg /= array.length
    let paddingNum = (kernel.length / 2) | 0

    for (let i = 0; i < paddingNum; i++) {
      copy.push(avg)
      copy.unshift(avg)
    }

    // i + paddingNum is offset index
    for (let i = 0; i < array.length; i++) {
      let j = i + paddingNum
      let sum = 0
      for (let k = j; k - j < kernel.length; k++) {
        sum += copy[k - paddingNum] * kernel[k - j]
      }
      sum /= norm
      retval.push(sum)
    }
    return retval
  }

  /**
   * @brief Estimates demand rate histogram using convolution with 1D Sobel Kernel
   * @returns Demand rate histogram
   */
  estOrderFreq(): number[] {
    let demandHistObj = this.getNumOrderByMonth()
    // Reverse needed as most recent months at lowest index
    demandHistObj = demandHistObj.reverse()
    let demandHist: number[] = []
    demandHistObj.forEach(element => {
      demandHist.push(element.orders)
    })
    return this.convolve1D(demandHist, [-1, -2, 0, 2, 1], 6)
  }
}
