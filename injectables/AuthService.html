import { Injectable } from '@angular/core';
import { 
  getAuth,
  onAuthStateChanged,
  signOut, 
  signInWithEmailAndPassword, 
  createUserWithEmailAndPassword 
} from "firebase/auth"

import { getFunctions, httpsCallable } from "firebase/functions";
import { LoadingService } from './loading.service';
import {authentication } from 'src/main';
import { Observable } from 'rxjs';
import { AlertService } from './alert.service';

const functions = getFunctions()

//const auth = getAuth()

const auth = authentication

onAuthStateChanged(auth, (user) => {
  if (user) {
    //user is signed in
    const uid = user.uid
  } else {
    //No user signed in
  }
})

@Injectable({
  providedIn: 'root'
})
export class AuthService {

  constructor(
    private loadingService: LoadingService,
    private alertService: AlertService
  ) {

   }

   async getCurrentUser(): Promise<any> {
    return new Promise(async (resolve, reject) => {
      const unsub = await auth.onAuthStateChanged(user => {
        resolve(user)
      }, reject)
    })
  }

   async signIn(email: string, password: string, pantry: string) {
    await this.loadingService.showLoader()
    await signInWithEmailAndPassword(auth, email, password).then(async (userCred) => {
      const isAdmin = httpsCallable(functions, 'isAdmin')
      // let token = await userCred.user.getIdToken()
      // let params = {
      //   idToken: token
      // }
      await isAdmin({pantry: pantry}).then((res) => {
        this.loadingService.dismissLoader()
        //res.data will be true or false
        let userIsAdmin = res.data
        if(!userIsAdmin) {
          this.signOut()
          this.alertService.presentAlert('Login Failed', 'Please check your credentials and try again later.', ['Ok'])
        }
      })
    }).catch((err) => {
      this.alertService.presentAlert('Login Failed', 'Please check your credentials and try again later.', ['Ok'])
      this.loadingService.dismissLoader()
    })
   }

   signOut() {
    signOut(auth)
   }

   static signOut() {
    signOut(auth)
   }

   async grantAdminAccess(uid: string, pantry: string) {
    const grantAdmin = httpsCallable(functions, 'makeAdmin')
    let params = {
      user: uid,
      pantryName: pantry
    }
    await grantAdmin(params).catch((err) => {
      this.alertService.presentAlert('Error', 'Could not modify permissions. Please try again later.', ['Ok'])
    })
   }

   async revokeAdminAccess(uid: string) {
    const revokeAdmin = httpsCallable(functions, 'revokeAdmin')
    let params = {
      user: uid
    }
    await revokeAdmin(params).catch((err) => {
      this.alertService.presentAlert('Error', 'Could not modify permissions. Please try again later.', ['Ok'])
    })
   }

}
