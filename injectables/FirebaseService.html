import { Injectable } from '@angular/core';

import { getFirestore, collection, addDoc, updateDoc, getDocs, getDoc, QuerySnapshot, DocumentSnapshot, doc, setDoc, deleteDoc } from 'firebase/firestore'

import { FoodItem } from '../../../../globalModels/foodItem'
import { Order } from '../../../../globalModels/order'
import { Pantry } from '../../../../globalModels/pantry'
import { OrderStatus } from '../../../../globalModels/orderStatus';

import { getFunctions, httpsCallable } from "firebase/functions";
import { User } from '../../../../globalModels/user';
import { LoadingService } from './loading.service';

const functions = getFunctions()

const db = getFirestore()

@Injectable({
  providedIn: 'root'
})
export class FirebaseService {

  //This will be used to determine which pantry instance is accessed
  //TODO: update/change this when the user logs in
  userId = ''
  pantryId = ''

  constructor(
    private loadingService: LoadingService
  ) {
    let temp = sessionStorage.getItem('uid')
    let temp2 = sessionStorage.getItem('pantryId')
    if (temp != null && temp2 != null) {
      this.userId = temp
      this.pantryId = temp2
    }
  }

  /**
   * Retrieves all pantries that are currently in the database
   * @returns A promise containing an array of all available Pantries
   */
  async getPantries(): Promise<Pantry[]> {
    await this.loadingService.showLoader()
    let snapshot = undefined
    try {
      snapshot = await getDocs(collection(db, 'pantries'))
    } catch (error) {
      this.loadingService.dismissLoader()
      throw error
    }
    if (snapshot == undefined) {
      this.loadingService.dismissLoader()
      throw Error('No Pantries Found')
    }
    var toReturn: Pantry[] = []
    snapshot.forEach((pantry) => {
      toReturn.push(this.pantryConv(pantry))
    })
    this.loadingService.dismissLoader()
    return toReturn
  }

  /**
   * Retrieves the pantry's DB id
   * @returns The pantry Id that is intialized on login
   */
  getPantryId() {
    return this.pantryId
  }

  /**
   * Retrieves Pantry Data from firebase database
   * @returns Promise that contains the Pantry Data from the database
   */
  async getPantry(): Promise<Pantry> {
    await this.loadingService.showLoader()
    let snapshot = undefined;
    try {
      snapshot = await getDoc(doc(db, 'pantries/' + this.pantryId))
    } catch (error) {
      this.loadingService.dismissLoader()
      throw (error)
    }
    if (snapshot == undefined) {
      this.loadingService.dismissLoader()
      throw Error('Food Pantry does not exist')
    }
    this.loadingService.dismissLoader()
    return this.pantryConv(snapshot)

  }

  /**
   * Converts Firebase Snapshot into Pantry Object
   * @param snapshot Firebase Document Snapshot of Pantry
   * @returns Pantry Object representaion of parameter
   */
  pantryConv(snapshot: DocumentSnapshot) {
    let data = snapshot.data()!
    return new Pantry(snapshot.id, data['name'], data['description'], data['address'], data['logoUrl'], data['primaryColor'], data['secondaryColor'], data['orderLimit'])
  }

  /**
   * Retrieves the inventory from the firebase database
   * @returns Promise containing an array of FoodItem Objects
   */
  async getInventory(): Promise<FoodItem[]> {
    await this.loadingService.showLoader()
    let snapshot = undefined;
    try {
      snapshot = await getDocs(collection(db, "pantries/" + this.pantryId + "/inventory/"))
    } catch (error) {
      //throw error to outer page
      this.loadingService.dismissLoader()
      throw error
    }
    if (snapshot == undefined) {
      this.loadingService.dismissLoader()
      throw Error('No Data Found')
    }
    
    let toReturn: FoodItem[] = []
    snapshot.forEach((doc) => {
      if (doc.exists()) {
        toReturn.push(this.foodItemConv(doc))
      }
    })
    this.loadingService.dismissLoader()
    return toReturn
  }

  /**
   * Converts Firebase Snapshot into Food Item Object
   * @param snap Firebase representation of a food item
   * @returns FoodItem object representaion of parameter
   */
  foodItemConv(snap: DocumentSnapshot) {
    let docData = snap.data()!
    return new FoodItem(snap.id, docData['name'], docData['quantity'], docData['orderQuan'], docData['limit'], docData['imageUrl'])
  }

  /**
   * Retrieves all orders for a pantry
   * @returns A Promise containing an array of Order Objects
   */
  async getOrders(): Promise<Order[]> {
    await this.loadingService.showLoader()
    let snapshot = undefined;
    try {
      snapshot = await getDocs(collection(db, "pantries/" + this.pantryId + "/orders/"))
    } catch (error) {
      this.loadingService.dismissLoader()
      //throw error to outer page
      throw error
    }
    if (snapshot == undefined) {
      this.loadingService.dismissLoader()
      throw Error('No Data Found')
    }

    let toReturn: Order[] = []

    for (let doc of snapshot.docs) {
      if (doc.exists()) {
        let toAdd = await this.orderConv(doc)
        toReturn.push(toAdd)
      }
    }
    this.loadingService.dismissLoader()
    // console.log('returning from orders')
    return toReturn
  }

  /**
   * Retireves items from a particular order
   * @param order DocumentSnapshot to retrives items from
   * @returns Promise containing an array of FoodItem Objects
   */
  async getOrderItems(order: DocumentSnapshot): Promise<FoodItem[]> {
    let snapshot = undefined
    let orderPath = order.ref.path
    try {
      snapshot = await getDocs(collection(db, orderPath + '/items'))
    } catch (error) {
      throw error
    }
    if (snapshot == undefined) {
      throw Error('No Data Found')
    }

    let toReturn: FoodItem[] = []
    snapshot.forEach((item) => {
      toReturn.push(this.foodItemConv(item))
    })
    // console.log('return from getOrderItems')
    return toReturn
  }

  /**
   * Converts Document Snapshot into an Order Object
   * @param snap DocumentSnapshot with order data
   * @returns Order Object that contains the data from the parameter
   */
  async orderConv(snap: DocumentSnapshot) {
    let docData = snap.data()!
    let items: FoodItem[] = []
    //TODO: actually pull for items for order
    items = await this.getOrderItems(snap)
    // console.log('return from order conv')
    return new Order(snap.id, docData['date'], docData['status'], docData['uid'], items)
  }

  /**
   * Adds a new item to the database
   * @param itemToAdd FoodItem to add to the database
   */
  async addItem(itemToAdd: FoodItem) {
    await this.loadingService.showLoader()
    let docRef = doc(db, 'pantries/' + this.pantryId + '/inventory/' + itemToAdd.id)
    let snapshot = await getDoc(docRef)
    if (snapshot.exists()) {
      console.log('calling-dismiss')
      this.loadingService.dismissLoader()
      throw Error('Item already exists')
    }
    try {
      await setDoc(docRef, {
        imageUrl: itemToAdd.imageUrl,
        limit: itemToAdd.limit,
        name: itemToAdd.name,
        quantity: itemToAdd.quantity
      })
    } catch (err) {
      console.log('calling-dismiss')
      this.loadingService.dismissLoader()
      throw err
    }
    console.log('calling-dismiss')
    this.loadingService.dismissLoader()
  }

  /**
   * Updates a FoodItem in the inventory
   * @param itemToUpdate FoodItem that is the item to update in the inventory
   */
  async updateInventoryItem(itemToUpdate: FoodItem) {
    await this.loadingService.showLoader()
    let docRef = doc(db, 'pantries/' + this.pantryId + '/inventory/' + itemToUpdate.id)
    let snapshot = await getDoc(docRef)
    if (snapshot.exists()) {
      await updateDoc(docRef, {
        name: itemToUpdate.name,
        limit: itemToUpdate.limit,
        quantity: itemToUpdate.quantity
      })
    } else {
      this.loadingService.dismissLoader()
      throw Error('Item does not exist!')
    }
    this.loadingService.dismissLoader()
  }

  /**
   * Removes a food item from the database
   * @param itemToRemove FoodItem to remove from the DB
   */
  async removeItem(itemToRemove: FoodItem) {
    await this.loadingService.showLoader()
    let docRef = doc(db, 'pantries/' + this.pantryId + '/inventory/' + itemToRemove.id)
    try {
      await deleteDoc(docRef)
    } catch (err) {
      this.loadingService.dismissLoader()
      throw err
    }
    this.loadingService.dismissLoader()
  }

  /**
   * Updates an order
   * @param order Order object to update in DB 
   * @param newStatus Integer status to change the order status to
   */
  async updateOrderStatus(order: Order, newStatus: OrderStatus) {
    await this.loadingService.showLoader()
    let docRef = doc(db, 'pantries/' + this.pantryId + '/orders/' + order.id)
    let snapshot = await getDoc(docRef)
    if (snapshot.exists()) {
      await updateDoc(docRef, {
        status: newStatus
      })
    } else {
      this.loadingService.dismissLoader()
      throw Error('Order does not exist!')
    }
    this.loadingService.dismissLoader()
  }

  /**
   * Updates Pantry details
   * @param pantry Pantry object to update in DB
   */
  async updatePantry(pantry: Pantry) {
    await this.loadingService.showLoader()
    let docRef = doc(db, 'pantries/' + this.pantryId)
    let snapshot = await getDoc(docRef)
    if (snapshot.exists()) {
      await updateDoc(docRef, {
        address: pantry.address,
        description: pantry.desc,
        logoUrl: pantry.logoUrl,
        name: pantry.name,
        orderLimit: pantry.orderLimit,
        primaryColor: pantry.primaryColor,
        secondaryColor: pantry.secondaryColor
      })
    } else {
      this.loadingService.dismissLoader()
      throw Error('pantry does not exist!')
    }
    this.loadingService.dismissLoader()
  }

  /**
   * Retrieves all users for the pantry that the user is currently logged into
   * @returns A promise containing an array of users
   */
  async getUsers(): Promise<User[]> {
    await this.loadingService.showLoader()
    const getUsers = httpsCallable(functions, 'getUsers')
    return getUsers().then(async (result: any) => {
      var userArr: User[] = []
      let users = result.data.users
      for (let user of users) {
        let toAdd = new User(user.uid, user.email, user.uid)
        toAdd.perms = await this.getPerms(toAdd)
        userArr.push(toAdd)
      }
      userArr.reverse()
      this.loadingService.dismissLoader()
      return userArr
    })
  }

  /**
   * Retrieves a user's premission from the authentication database
   * @param user User to retrieve permissions for
   * @returns An dictionary containing the user's custom claims in firebase
   */
  async getPerms(user: User) {
    console.log('calling getPerms')
    const getPerms = httpsCallable(functions, 'getPerms')
    let params = {
      email: user.email
    }
    return getPerms(params).then((res: any) => {
      let perms = res.data
      return perms
    })
  }

  /**
   * Grants a specific user admin perms
   * @param uid UID if the user to be promoted
   * @returns the result of the makeAdmin cloud function
   */
  async promoteUser(uid: string) {
    await this.loadingService.showLoader()
    const makeAdmin = httpsCallable(functions, 'makeAdmin')
    let params = {
      user: uid,
      pantryName: this.pantryId
    }
    return makeAdmin(params).then(() => {
      console.log('done with edit')
      this.loadingService.dismissLoader()
    })
  }

  /**
   * Removes a specific user's admin perms
   * @param uid UID of user to be demoted
   * @returns result of revokeAdmin cloud function
   */
  async demoteUser(uid: string) {
    await this.loadingService.showLoader()
    const revokeAdmin = httpsCallable(functions, 'revokeAdmin')
    let params = {
      user: uid
    }
    return revokeAdmin(params).then(() => {
      this.loadingService.dismissLoader()
    })
  }

}
