import { AfterViewInit, ViewChild } from '@angular/core';
import { Component, OnInit } from '@angular/core';
import { MatMenuTrigger } from '@angular/material/menu';
import { MatPaginator } from '@angular/material/paginator';
import { MatSort } from '@angular/material/sort';
import { MatTableDataSource } from '@angular/material/table';
import {MatDialog, } from '@angular/material/dialog';
import { AlertController, } from '@ionic/angular';
import { FoodItem } from '../../../../globalModels/foodItem';
import { FirebaseService } from '../services/firebase.service';
import { Inject } from '@angular/core';
import {MAT_DIALOG_DATA} from '@angular/material/dialog';
import { LoadingService } from '../services/loading.service';




@Component({
  selector: 'inventory-management',
  templateUrl: './inventory-management.page.html',
  styleUrls: ['./inventory-management.page.scss'],
})
export class InventoryManagementPage implements AfterViewInit{
  

  displayedColumns: string[] = ['name', 'quantity', 'limit', 'actions'];
  records: MatTableDataSource<FoodItem>
  length: number = 0;
  pageSize = 5;
  pageSizeOptions: number[] = [5, 10, 25, 100];

  @ViewChild(MatPaginator) paginator: MatPaginator;
  @ViewChild(MatSort) sort: MatSort;
  @ViewChild(MatMenuTrigger) actions: MatMenuTrigger;

  static addDialogRef: any

  constructor(
    public dialog: MatDialog,
    private alertController: AlertController,
    private firebaseService: FirebaseService,
    ) { }

  async ngOnInit() {    
    
  }

  async updateTableSource() {
    let data = await this.firebaseService.getInventory()
    this.records = new MatTableDataSource<FoodItem>(data)
    this.length = this.records.data.length;
    this.records.paginator = this.paginator;
    this.records.sort = this.sort;
  }
  async ngAfterViewInit() {
    await this.updateTableSource()
  }

  updateFilter(event: Event){
    const filterValue = (event.target as HTMLInputElement).value;
    this.records.filter = filterValue.trim().toLowerCase();

    if(this.records.paginator) {
      this.records.paginator.firstPage();
    }
  }
  async removeItem(item: FoodItem) {
    //find the index from id
    await this.firebaseService.removeItem(item)
    let removed = this.records.data.splice(this.records.data.indexOf(item), 1)
    if(this.records.paginator) {
      this.records.paginator.firstPage();
    }
    this.records._updateChangeSubscription();
  }
 
  async openAlert(item: FoodItem){
    const alert = await this.alertController.create({
      header: 'Alert',
      message: 'Are you sure you want to remove this item?',
      buttons: [{
        text: 'Cancel',
        role: 'cancel',
        handler: () => {}
      },
      {
        text: 'Delete Item',
        role: 'confirm',
        handler: () => {
          this.removeItem(item)
        }
      }],
    });
    await alert.present();
  }

   openEditDialog(item: FoodItem){
    const dialogRef = this.dialog.open(EditDialog, {restoreFocus: false, data: item});
    dialogRef.afterClosed().subscribe( () => {
      this.actions.focus()
      this.records._updateChangeSubscription();
    });
  }

   async openAddDialog(){
    const dialogRef = this.dialog.open(AddDialog, {restoreFocus: false});
    InventoryManagementPage.addDialogRef = dialogRef
    dialogRef.afterClosed().subscribe( () => {
      this.updateTableSource()
      this.actions.focus()
      this.records._updateChangeSubscription();
    })
  }

  static async closeAddDialog() {
    await InventoryManagementPage.addDialogRef.close()
  }

}

@Component({

})
export class AddDialog implements OnInit {
  nameInput: string
  quantityInput: number
  limitInput: number
  urlInput: string
  forceDisable = false
  constructor(
    private firebaseService: FirebaseService,
    private loadingService: LoadingService
    ){}
  ngOnInit(): void {
      
  }

  /**
   * Adds item to the UI table and the DB
   */
  async AddItem(){
    this.forceDisable = true
    //write to db or something idk
    if(!this.limitInput) {
      this.limitInput = 0
    }
    let item: FoodItem = {
      id: this.nameInput.toLowerCase(),
      name: this.nameInput,
      quantity: this.quantityInput,
      limit: this.limitInput,
      imageUrl: this.urlInput
    }
    console.log(item)
    await this.firebaseService.addItem(item).then(() => {
      InventoryManagementPage.closeAddDialog()
      this.forceDisable = false
    }).catch((err) => {
      this.forceDisable = false
    })
    this.forceDisable = false
  }
}

@Component({
  selector: 'edit-dialog',
  templateUrl: 'edit-dialog.html'
})
export class EditDialog implements OnInit{
  constructor(@Inject(MAT_DIALOG_DATA) public data: FoodItem,
  private firebaseService: FirebaseService){}
  private cancelItem = 0;
  private cancelLimit = 0;
  //private name = this.data.name;
  nameInput: string = this.data.name
  ngOnInit() {
    console.log(this.data)
  }

  /**
   * increments item quantity
   * @param item FoodItem to increment
   */
  addItem(item: FoodItem){
    item.quantity = item.quantity + 1;
    this.cancelItem = this.cancelItem - 1;
  }

  /**
   * decrements item quantity
   * @param item FoodItem to decrement
   */
  removeItem(item: FoodItem){
    item.quantity = item.quantity - 1;
    this.cancelItem = this.cancelItem + 1;
  }

  /**
   * increments and items's limit
   * @param item FoodItem that will have limit modifed
   */
  addLimit(item: FoodItem){
    if(item.limit){
      item.limit = item.limit + 1
      this.cancelLimit = this.cancelLimit - 1
    } else {
      item.limit = 1
      this.cancelLimit = this.cancelLimit - 1
    }
  }

  /**
   * decrements and item's limit
   * @param item FoodItem that will have limit modified
   */
  removeLimit(item: FoodItem){
    if(item.limit){
      item.limit = item.limit - 1;
      this.cancelLimit = this.cancelLimit + 1
    } else if(item.limit == 1) {
      item.limit = undefined;
      this.cancelLimit = this.cancelLimit + 1
    }
  }

  /**
   * Updates DB info for an item. Called when item changes are saved
   * @param item FoodItem to update in database
   */
  async saveData(item: FoodItem){
    //write to db
    console.log(this.nameInput)
    if(this.nameInput != ""){
      item.name = this.nameInput
    }
    await this.firebaseService.updateInventoryItem(item)
  }

  /**
   * Discards changes made to food item. Called when Cancel is pressed
   * @param item FoodItem to discard changes for
   */
  cancelData(item: FoodItem){
    item.quantity = item.quantity + this.cancelItem;
    if(item.limit){
      item.limit = item.limit + this.cancelLimit
      if(item.limit == 0){
        item.limit = undefined
      }
    }
  }
}
