import { Component, ElementRef, Inject, OnInit, ViewChild } from '@angular/core';
import { Pantry } from '../../../../globalModels/pantry'
import { FirebaseService } from '../services/firebase.service';
import { MAT_DIALOG_DATA, MatDialog } from '@angular/material/dialog';
import { MatMenuTrigger } from '@angular/material/menu';
import { IonContent } from '@ionic/angular';
import { AlertService } from '../services/alert.service';

@Component({
  selector: 'app-profile',
  templateUrl: './profile.page.html',
  styleUrls: ['./profile.page.scss'],
})
export class ProfilePage implements OnInit {

  pantry: Pantry = new Pantry('id', 'name', 'desc', 'address', 'logo', 'red', 'black', 2, [], [], [])
  @ViewChild (MatMenuTrigger) actions: MatMenuTrigger

  @ViewChild('iContent', {static: true})
  private iContentRef: ElementRef<IonContent>
  private iConStyle: CSSStyleDeclaration
  constructor(
    private firebase: FirebaseService,
    public dialog: MatDialog,
    private alertService: AlertService
  ) { }

  /**
   * Gets pantry data on page init
   */
  async ngOnInit() {
    try {
      this.pantry = await this.firebase.getPantry()
      console.log(this.pantry)
      this.updateColors()
    } catch (err) {
      this.alertService.presentAlert('Error', 'Could not retrieve data. Please try again later.', ['Ok'])
    }
  }

  /**
   * Opens the Pantry edit dialog
   * @param data Pantry to Edit
   */
  openEditDetailsDialog(data: Pantry){
    const dialogRef = this.dialog.open(EditDetailsDialog, {restoreFocus: false, data: data});
    dialogRef.afterClosed().subscribe(() => {
      this.actions.focus()
    });
  }

  /**
   * Updates UI Colors
   */
  updateColors(){
    document.body.style.setProperty('--ion-color-primary', this.pantry.primaryColor)
    document.body.style.setProperty('--ion-color-primary-contrast', this.pantry.secondaryColor)
  }

}

@Component({
  selector: 'edit-details-dialog',
  templateUrl: 'edit-details-dialog.html'
}) export class EditDetailsDialog implements OnInit {
  public name: string = this.data.name
  public description: string = this.data.desc
  public address: string = this.data.address
  public icon: string = this.data.logoUrl
  public primary: string = this.data.primaryColor
  public secondary: string = this.data.secondaryColor
  public maxOrder: number = this.data.orderLimit

  constructor(@Inject(MAT_DIALOG_DATA) public data: Pantry,
    private firebaseService: FirebaseService,
    private alertService: AlertService
  ){}

  ngOnInit(): void {
    // console.log(this.data)
  }

  /**
   * Saves the pantry data that is changed in the edit dialog
   */
  async saveData(){
    let pantry: Pantry = {
      "name": this.name,
      "id": this.data.id,
      "desc": this.description,
      "address": this.address,
      "logoUrl": this.icon,
      "primaryColor": this.primary,
      "secondaryColor": this.secondary,
      "orderLimit": this.maxOrder
    }
    this.data.name = this.name
    this.data.desc = this.description
    this.data.address = this.address
    this.data.logoUrl = this.icon
    this.data.primaryColor = this.primary
    this.data.secondaryColor = this.secondary
    this.data.orderLimit = this.maxOrder
    try {
      await this.firebaseService.updatePantry(pantry)
      this.updateColors()
    } catch (err) {
      this.alertService.presentAlert('Error', 'Could not update data. Please try again later.', ['Ok'])
    }
    
  }

  /**
   * Updates UI Colors
   */
  updateColors(){
    document.body.style.setProperty('--ion-color-primary', this.data.primaryColor)
    document.body.style.setProperty('--ion-color-primary-contrast', this.data.secondaryColor)
  }

}
