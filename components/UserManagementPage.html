import { Component, OnInit, ViewChild } from '@angular/core';
import { MatTableDataSource } from '@angular/material/table';
import { User } from '../../../../globalModels/user';
import { MatPaginator } from '@angular/material/paginator';
import { MatSort } from '@angular/material/sort';
import { MatMenuTrigger } from '@angular/material/menu';
import { MatDialog } from '@angular/material/dialog';
import { AlertController } from '@ionic/angular';
import { FirebaseService } from '../services/firebase.service';
import { AlertService } from '../services/alert.service';

@Component({
  selector: 'app-user-management',
  templateUrl: './user-management.page.html',
  styleUrls: ['./user-management.page.scss'],
})
export class UserManagementPage implements OnInit {

  displayedColumns: string[] = ['email', 'permissions', 'actions']
  records: MatTableDataSource<User>
  length: number = 0
  pageSize = 5
  pageSizeOptions: number[] = [5, 10, 25, 100]
  pantryId = ''

  @ViewChild(MatPaginator) paginator: MatPaginator
  @ViewChild(MatSort) sort: MatSort
  @ViewChild(MatMenuTrigger) actions: MatMenuTrigger

  constructor(
    public dialog: MatDialog,
    private alertController: AlertController,
    private firebaseService: FirebaseService,
    private alertService: AlertService
  ) {

  }

  /**
   * Retreives a pantry's users and intializes the table data
   */
  async ngOnInit() {
    try {
      this.pantryId = this.firebaseService.getPantryId()
      let data = await this.firebaseService.getUsers()
      this.records = new MatTableDataSource<User>(data)
      this.length = this.records.data.length
      this.records.paginator = this.paginator
      this.records.sort = this.sort
    } catch (err) {
      this.alertService.presentAlert('Error', 'Could not retrieve data. Please try again later.', ['Ok'])
    }
    
    // for (let d of data) {
    //   console.log(d)
    // }
  }

  /**
   * Calls the firebase service to promote a user. Then updates user perms on the table
   * @param user user to promote
   */
  async makeAdmin(user: User) {
    await this.firebaseService.promoteUser(user.uid).then(async () => {
      console.log('getPerms')
      user.perms = await this.firebaseService.getPerms(user)
      console.log(user.perms)
    })
  }

  /**
   * Calls the firebase service to demote a user. Then updates user perms on the table.
   * @param user user to demote
   */
  async removeAdmin(user: User) {
    await this.firebaseService.demoteUser(user.uid).then(async () => {
      user.perms = await this.firebaseService.getPerms(user)
    })
  }

}
