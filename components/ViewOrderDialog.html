import { AfterViewInit, ChangeDetectorRef, Component, OnInit, ViewChild } from '@angular/core';
import { MatPaginator } from '@angular/material/paginator';
import { MatSort } from '@angular/material/sort';
import { MatTable, MatTableDataSource } from '@angular/material/table';
import { FirebaseService } from '../services/firebase.service';
import { Order } from '../../../../globalModels/order';
import { FoodItem } from '../../../../globalModels/foodItem';
import { OrderStatus } from '../../../../globalModels/orderStatus';
import { MatDialog, MatDialogRef } from '@angular/material/dialog';
import { AlertController } from '@ionic/angular';
import { Inject } from '@angular/core';
import { MAT_DIALOG_DATA } from '@angular/material/dialog';
import { MatMenuTrigger } from '@angular/material/menu';
import { Observable, from } from 'rxjs';

@Component({
  selector: 'app-order-management',
  templateUrl: './order-management.page.html',
  styleUrls: ['./order-management.page.scss'],
})

export class OrderManagementPage implements AfterViewInit {

  displayedColumns: string[] = ['date', 'id', 'status', 'actions'];
  records: MatTableDataSource<Order> = new MatTableDataSource<Order>([]);

  length: number = 0;
  pageSize = 5;
  pageSizeOptions: number[] = [5, 10, 25, 100];

  @ViewChild(MatTable, { static: true }) table: MatTable<any>;
  @ViewChild(MatPaginator) paginator: MatPaginator;
  @ViewChild(MatSort) sort: MatSort
  @ViewChild(MatMenuTrigger) actions: MatMenuTrigger;
  constructor(
    public dialog: MatDialog,
    private firebase: FirebaseService,
    private alertController: AlertController,
  ) { }

  ngOnInit() {

  }

  async ionViewWillEnter() {

  }

  async ngAfterViewInit() {
    let orders = await this.firebase.getOrders()
    
    this.records.data = orders
    this.length = this.records.data.length
    this.records.paginator = this.paginator;
    this.records.sort = this.sort;
  }


  updateFilter(event: Event) {
    const filterValue = (event.target as HTMLInputElement).value;
    this.records.filter = filterValue.trim().toLowerCase();

    if (this.records.paginator) {
      this.records.paginator.firstPage();
    }
  }

  getStatus(status: OrderStatus) {
    return OrderStatus[status]
  }

  removeItem(order: Order) {
    //find the index from id
    let removed = this.records.data.splice(this.records.data.indexOf(order), 1)
    if (this.records.paginator) {
      this.records.paginator.firstPage();
    }
    this.records._updateChangeSubscription();
  }

  async openAlert(order: Order) {
    const alert = await this.alertController.create({
      header: 'Alert',
      message: 'Are you sure you want to remove this item?',
      buttons: [{
        text: 'Cancel',
        role: 'cancel',
        handler: () => { }
      },
      {
        text: 'Delete Item',
        role: 'confirm',
        handler: () => {
          this.removeItem(order)
        }
      }],
    });
    await alert.present();
  }

  openEditDialog(order: Order) {
    const dialogRef = this.dialog.open(EditOrderDialog, { restoreFocus: false, data: order })
    dialogRef.afterClosed().subscribe(() => {
      this.actions.focus()
      this.records._updateChangeSubscription();
    });
  }

  openViewDialog(order: Order) {
    const dialogRef = this.dialog.open(ViewOrderDialog, { restoreFocus: false, data: order })
    dialogRef.afterClosed().subscribe(() => {
      this.actions.focus()
      this.records._updateChangeSubscription();
    });
  }
}
