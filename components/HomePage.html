import { Component } from '@angular/core';
import { FirebaseService } from '../services/firebase.service';
import { Pantry } from '../../../../globalModels/pantry';

/**
 * Landing page for the admin app
 */
@Component({
  selector: 'app-home',
  templateUrl: 'home.page.html',
  styleUrls: ['home.page.scss'],
})
export class HomePage {
  pantry: Pantry
  logoUrl: string
  constructor(
    private firebaseService: FirebaseService
  ) {}

  /**
   * Init method retrieves the pantry so the patry name and logo can be set
   */
    async ngOnInit() {
      this.pantry = await this.firebaseService.getPantry()
      this.logoUrl = this.pantry.logoUrl
    }

}
