import { Component, OnInit, ViewChild } from '@angular/core';
import { ChartConfiguration, ChartOptions, ChartType } from "chart.js";
import { ForecastService } from '../services/forecast.service';
import { FirebaseService } from '../services/firebase.service';
import { FoodItem } from '../../../../globalModels/foodItem';
import { AlertService } from '../services/alert.service';


interface TimestampedOrders {
  orders: number
  month: string
}

interface TimestampedItems {
  items: number
  month: string
}
@Component({
  selector: 'app-pantry-statistics',
  templateUrl: './pantry-statistics.page.html',
  styleUrls: ['./pantry-statistics.page.scss'],
})
export class PantryStatisticsPage implements OnInit {
  fromDate = ""
  endDate = ""
  inventory: FoodItem[]
  selectedFoodItem: string
  ordersByMonth: TimestampedOrders[]
  ordersByMonthMonths: string[] = []
  ordersByMonthNums: number[] = []
  itembyMonth: TimestampedItems[]
  itembyMonthMonths: string[] = []
  itembyMonthNums: number[] = []

  /**
   * Separates data into arrays for display in charts
   */
  convertItemByMonth() {
    this.itembyMonth.forEach(element => {
      this.itembyMonthMonths.push(element.month)
      this.itembyMonthNums.push(element.items)
    })
  }

  /**
   * DEBUG method
   */
  fromDateChange() {
    // console.log(this.fromDate)
  }

  /**
   * DEBUG method
   */
  endDateChange() {
    // console.log(this.endDate)
  }

  /**
   * DEBUG method
   */
  selectChange() {
    // console.log(this.selectedFoodItem)
  }

  /**
   * Separates data into arrays for display in charts
   */
  convertOrdersByMonth() {
    this.ordersByMonth.forEach(element => {
      this.ordersByMonthMonths.push(element.month)
      this.ordersByMonthNums.push(element.orders)
    })
  }

  /**
   * Updates the item by month chart with proper data.
   */
  updateItemChart() {
    this.lineChartDataItem = {
      labels: this.itembyMonthMonths,
      datasets: [
        {
          data: this.itembyMonthNums,
          label: this.selectedFoodItem?this.selectedFoodItem:"",
          fill: true,
          tension: .5,
          borderColor: 'black',
          backgroundColor: 'rgba(0,255,0,0.3)'
        }
      ]
    }
  }

  /**
   * Updates data when the user presses submit on "item by month" graph
   */
  updateItemGraph() {
    this.itembyMonth = []
    this.itembyMonthNums = []
    this.itembyMonthMonths = []
    console.log("submit clicked")
    if(this.selectedFoodItem){
      let fromDateObj = new Date(this.fromDate)
      let endDateObj
      if(this.endDate) {
        endDateObj = new Date(this.endDate)
      } else {
        endDateObj = new Date()
      }
      console.log(fromDateObj.getMonth())
      console.log(endDateObj.getMonth())
      
      if(fromDateObj.getMonth() < endDateObj.getMonth()) {
        //different method but for now doing same
        console.log("earlier start date")
        this.itembyMonth = this.forecaster.getItemDemandInRange(this.selectedFoodItem, fromDateObj, endDateObj)
        this.convertItemByMonth()
      } else {
        this.itembyMonth = this.forecaster.getItemDemandByMonth(this.selectedFoodItem)
        this.convertItemByMonth()
      }
      this.updateItemChart()
    }
    
  }
  public lineChartDataOrders: ChartConfiguration<'line'>['data']
  public lineChartDataItem: ChartConfiguration<'line'>['data']
  public lineChartOptions: ChartOptions<'line'> = {
    responsive: true
  };
  public lineChartLegend = true;

  constructor(
    private forecaster: ForecastService, 
    private firebaseService: FirebaseService,
    private alertService: AlertService
    ) { }
  
  /**
   * Initializes Charts and Data
   */
  async ngOnInit() {
    try {
      let pantry = await this.firebaseService.getPantry()
      this.inventory = await this.firebaseService.getInventory()
      let orders = await this.firebaseService.getOrders()
      this.forecaster.setData(pantry, this.inventory, orders)
    } catch (err) {
      this.alertService.presentAlert('Error', 'Could not retrieve data. Please try again later.', ['Ok'])
    }
    
    

   
    console.log("Getting number of orders by month")
    this.ordersByMonth = this.forecaster.getNumOrderByMonth()
    console.log(this.forecaster.getNumOrderByMonth())
    this.convertOrdersByMonth()
    this.lineChartDataOrders = {
      labels: this.ordersByMonthMonths,
      datasets: [
        {
          data:this.ordersByMonthNums,
          label: 'Orders',
          fill: true,
          tension: 0.5,
          borderColor: 'black',
          backgroundColor: 'rgba(255,0,0,0.3)'
        }
      ]
    };

    this.lineChartDataItem = {
      labels: this.itembyMonthMonths,
      datasets: [
        {
          data: this.itembyMonthNums,
          label: this.selectedFoodItem?this.selectedFoodItem:"",
          fill: true,
          tension: .5,
          borderColor: 'black',
          backgroundColor: 'rgba(0,255,0,0.3)'
        }
      ]
    }
    console.log("Searching for bread demand by month")
    console.log(this.forecaster.getItemDemandByMonth("bread"))

  }

}
